\subsection{High-agreement tangent-star barrier}
\label{sec:tangent-star-barrier}

\textbf{Status: \textsc{New-local} finite MDS theorem.}
This subsection isolates the high-agreement regime in which the elementary
``moving root'' or tangent-star obstruction is not only present but optimal.
It is a support-wise finite-slope statement; projective or vertical-slope
samplers must add their own endpoint convention.

For a received line \(\ell_z=f+zg\), write \(\mathcal B_a(f,g)\) for the set
of finite slopes \(z\in F\) for which there are a support \(S\subseteq D\),
\(|S|\ge a\), and a codeword \(c_z\in C\) such that \(f+zg=c_z\) on \(S\),
while the same support is support-wise noncontained, i.e. there are no
\(c_f,c_g\in C\) with \(f=c_f\) and \(g=c_g\) on \(S\).  Thus
\[
  \operatorname{LD}_{\rm sw}(C,a)
  =\sup_{f,g:D\to F}|\mathcal B_a(f,g)|
\]
under the finite-slope convention.

\begin{theorem}[tangent-star lower bound and high-agreement optimality]
\label{thm:tangent-star-exact}
Let \(C\subseteq F^D\) be an MDS linear code of length \(n\) and dimension
\(k\).  Let \(k+1\le a\le n\).  Then
\[
  \operatorname{LD}_{\rm sw}(C,a)\ge n-a+1.
\]
If, in addition,
\[
  3a-2n\ge k,
\]
then every received line has at most \(n-a+1\) support-wise noncontained finite
bad slopes at agreement \(a\).  Hence
\[
  \operatorname{LD}_{\rm sw}(C,a)=n-a+1
  \qquad\text{whenever }k+1\le a\le n\text{ and }3a-2n\ge k.
\]
Moreover, if equality holds and \(n-a+1\ge2\), then every extremal line is a
tangent star: there are codewords \(c_f,c_g\in C\) and a set
\(S_0\subseteq D\) of size \(a-1\) such that \(f=c_f\) and \(g=c_g\) on
\(S_0\), and, writing \(\Omega=D\setminus S_0\),
\[
  x\longmapsto
  -\frac{f(x)-c_f(x)}{g(x)-c_g(x)}
\]
 is a bijection from \(\Omega\) to \(\mathcal B_a(f,g)\).  In particular, each
bad slope is witnessed on exactly the common support \(S_0\) plus its unique
moving residual root.
\end{theorem}

\begin{proof}
For the lower bound, choose a set \(S_0\subseteq D\) of size \(a-1\), put
\(\Omega=D\setminus S_0\), and choose distinct finite values
\(\theta_x\in F\) for \(x\in\Omega\).  This is possible because
\(|\Omega|=n-a+1\le n\le |F|\).  Let \(f=g=0\) on \(S_0\), and for
\(x\in\Omega\) set
\[
  g(x)=1,
  \qquad
  f(x)=-\theta_x .
\]
For the slope \(z=\theta_x\), the received word \(f+zg\) agrees with the zero
codeword on \(S_0\cup\{x\}\), a set of size \(a\).  The same support is
noncontained: if both \(f\) and \(g\) were restrictions of codewords on it,
then, since \(|S_0|=a-1\ge k\), those codewords would have to be zero, but
\(g(x)=1\).  Thus the \(|\Omega|=n-a+1\) slopes \(\theta_x\) are bad.

For the upper bound, let \(M=|\mathcal B_a(f,g)|\).  If \(M\le1\), there is
nothing to prove.  Otherwise choose two distinct bad slopes \(z_1,z_2\), with
witnessing supports \(S_1,S_2\) and codewords \(c_1,c_2\).  On
\(I=S_1\cap S_2\),
\[
  f+z_i g=c_i\qquad (i=1,2),
\]
so, because \(z_1\ne z_2\), both \(f\) and \(g\) agree on \(I\) with the
codewords
\[
  c_g=\frac{c_1-c_2}{z_1-z_2},
  \qquad
  c_f=\frac{z_1c_2-z_2c_1}{z_1-z_2}.
\]
Let \(S_0\) be the maximal common code-line support for this pair
\((c_f,c_g)\), and put \(b=|S_0|\).  Since
\[
  |I|\ge |S_1|+|S_2|-n\ge2a-n,
\]
we have \(b\ge2a-n\), hence
\[
  a+b-n\ge3a-2n\ge k.
\]
Apply Proposition~\ref{prop:common-residual} with this maximal \(S_0\).  The
maximality makes the common residual count \(c_0\) equal to zero.  Therefore
all support-wise noncontained bad slopes satisfy
\[
  M\le
  \left\lfloor
    \frac{n-b}{\max(1,a-b)}
  \right\rfloor .
\]
If \(b\ge a\), this is at most \(n-b\le n-a\).  If \(b<a\), write
\(d=a-b\ge1\) and \(r=n-a\).  Then
\[
  M\le \left\lfloor\frac{r+d}{d}\right\rfloor
    =1+\left\lfloor\frac r d\right\rfloor
    \le r+1=n-a+1.
\]
This proves optimality.

Assume now that equality holds and \(n-a+1=r+1\ge2\).  The displayed bound
cannot be sharp with \(b\ge a\), and if \(b<a\) it can be sharp only when
\(d=a-b=1\).  Thus \(|S_0|=a-1\).  Proposition~\ref{prop:common-residual}
then charges each bad slope to at least one coordinate of
\(\Omega=D\setminus S_0\), and each coordinate can vanish for at most one
finite slope.  Since both the number of bad slopes and the number of residual
coordinates are \(r+1\), every residual coordinate is used exactly once, no
coordinate has zero direction residual, and the displayed residual-root map is
a bijection onto \(\mathcal B_a(f,g)\).  Each bad support must therefore be
\(S_0\) together with the unique residual root for its slope.
\end{proof}

\begin{corollary}[no seventh non-tangent slope past agreement 506]
\label{cor:no-nontangent-past-506}
Let \(F=\F_{17^{32}}\), let \(H\subset F^*\) have order \(512\), and let
\[
  C=\RS[F,H,256].
\]
Under the finite-slope support-wise MCA convention,
\[
  \operatorname{LD}_{\rm sw}(C,a)=513-a
  \qquad\text{for every }a\ge427.
\]
In particular,
\[
  \operatorname{LD}_{\rm sw}(C,506)=7,
  \qquad
  \operatorname{LD}_{\rm sw}(C,507)=6.
\]
Since
\[
  6\cdot2^{128}<17^{32}<7\cdot2^{128},
\]
agreement \(507\) is below the \(2^{-128}\) bad-slope budget, while agreement
\(506\) is already above it.  Equivalently, on the integer Hamming-radius grid,
radius \(5\) is safe from finite-slope support-wise MCA and radius \(6\) is
not.  In real-radius notation the endpoint lies between
\(5/512\) and \(6/512\); the convention for closed balls determines which
endpoint is printed.
\end{corollary}

\begin{proof}
Here \(n=512\), \(k=256\), and the high-agreement condition is
\[
  3a-2n\ge k
  \quad\Longleftrightarrow\quad
  3a-1024\ge256
  \quad\Longleftrightarrow\quad
  a\ge427.
\]
Theorem~\ref{thm:tangent-star-exact} gives the exact staircase
\(513-a\) in this range.  The arithmetic inequalities follow by direct integer
calculation:
\[
  17^{32}-6\cdot2^{128}
  =326217393234836465063858652730341978625>0,
\]
while
\[
  7\cdot2^{128}-17^{32}
  =14064973686101998399515954701426232831>0.
\]
Thus six finite bad slopes have density at most \(2^{-128}\), and seven have
density strictly larger than \(2^{-128}\).
\end{proof}

\begin{remark}[what this settles and what it does not]
Corollary~\ref{cor:no-nontangent-past-506} rules out any mechanism that would
add a seventh finite support-wise bad slope at agreements \(a\ge507\) for the
\(\F_{17^{32}},n=512,k=256\) row.  It does not by itself prove a CA,
curve-MCA, projective-slope, interleaved-list, or protocol theorem.  Those
objects still require their own conversion and field-ledger checks.
\end{remark}
